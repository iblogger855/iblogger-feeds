# សេចក្តីផ្តើមអំពីទីផ្សារហិរញ្ញវត្ថុសកល (Introduction to Global Financial Markets) (កម្មវិធីសិក្សាផ្នែកសិល្បៈសេរីរបស់សហរដ្ឋអាមេរិក - US Liberal Arts Curriculum)

សេចក្តីផ្តើមអំពីទីផ្សារហិរញ្ញវត្ថុសកល (Introduction to Global Financial Markets) ត្រូវបានសិក្សានៅឆ្នាំទីមួយនៃជំនាញពាណិជ្ជកម្មសកល (Global Commerce) ជាទូទៅទន្ទឹមនឹងមុខវិជ្ជា គោលការណ៍សេដ្ឋកិច្ចមីក្រូ (Principles of Microeconomics)។ មុខវិជ្ជានេះមិនតម្រូវឱ្យមានចំណេះដឹងផ្នែកហិរញ្ញវត្ថុ ឬធុរកិច្ចពីមុនមកនោះទេ។ មុខវិជ្ជានេះមានវត្តមានឡើងដើម្បីផ្តល់ឱ្យនិស្សិតនូវទិដ្ឋភាពទូទៅនៃរបៀបដែលលុយកាក់ ដើមទុន និងទំនិញផ្លាស់ទីឆ្លងកាត់ព្រំដែនជាតិ — ដោយបង្កើតជាវាក្យសព្ទស្ថាប័ន (ផ្សារភាគហ៊ុន, IMF, ធនាគារពិភពលោក, WTO, សាជីវកម្មពហុជាតិ) ដែលរាល់មុខវិជ្ជាហិរញ្ញវត្ថុ និងធុរកិច្ចអន្តរជាតិនាពេលខាងមុខសន្មតថានិស្សិតមានរួចជាស្រេចហើយ។ សម្រាប់ផ្នែកនិរន្តរភាពនៃជំនាញទ្វេនេះ មុខវិជ្ជានេះគឺមានសារៈសំខាន់ដូចគ្នា៖ ការយល់ដឹងអំពីកន្លែងដែលដើមទុនហូរទៅ នឹងកំណត់ថាតើគម្រោងនិរន្តរភាពណាខ្លះដែលទទួលបានហិរញ្ញប្បទាន ហើយគម្រោងណាខ្លះដែលមិនទទួលបាន។

## ១. អ្វីទៅជាទីផ្សារហិរញ្ញវត្ថុ (What Financial Markets Are)

* **ទីផ្សារមូលបត្រកម្មសិទ្ធិ (Equity Markets)**
    * *ការផ្តោតលើ (Focus)៖* ទីផ្សារដែលកម្មសិទ្ធិនៅក្នុងក្រុមហ៊ុន (ភាគហ៊ុន) ត្រូវបានទិញនិងលក់។ និស្សិតសិក្សាពីរបៀបដែលក្រុមហ៊ុនចុះបញ្ជីនៅលើផ្សារភាគហ៊ុនតាមរយៈការបោះផ្សាយលក់មូលបត្រជាសាធារណៈលើកដំបូង (Initial Public Offerings - IPOs) របៀបដែលថ្លៃភាគហ៊ុនឆ្លុះបញ្ចាំងពីការរំពឹងទុកអំពីប្រាក់ចំណូលនាពេលអនាគត និងរបៀបដែលទីផ្សារភាគហ៊ុនបែងចែកដើមទុននៅក្នុងសេដ្ឋកិច្ច។ ការវិនិយោគលើបរិស្ថាន សង្គម និងអភិបាលកិច្ច (Environmental, Social, and Governance - ESG) ត្រូវបានណែនាំជាករណីសិក្សាអំពីរបៀបដែលព័ត៌មានមិនមែនហិរញ្ញវត្ថុចូលទៅក្នុងទីផ្សារភាគហ៊ុន។
* **ទីផ្សារបំណុល ឬទីផ្សារសញ្ញាបណ្ណ (Debt Markets - Bond Markets)**
    * *ការផ្តោតលើ (Focus)៖* ទីផ្សារដែលអ្នកខ្ចី (រដ្ឋាភិបាល សាជីវកម្ម) ប្រមូលប្រាក់កម្ចីដោយការបោះផ្សាយសញ្ញាបណ្ណ — ការសន្យាថានឹងសងប្រាក់ដើមវិញ រួមជាមួយការប្រាក់។ និស្សិតសិក្សាអំពីទំនាក់ទំនងច្រាសគ្នារវាងថ្លៃសញ្ញាបណ្ណ និងអត្រាការប្រាក់ ការវាយតម្លៃឥណទាន (Credit Ratings) និងបំណុលរដ្ឋ (Sovereign Debt)។ សញ្ញាបណ្ណបៃតង (Green Bonds) — ឧបករណ៍បំណុលដែលប្រាក់ចំណូលបានមកពីការលក់ត្រូវបានរឹតត្បិតសម្រាប់តែប្រើប្រាស់លើគម្រោងដែលផ្តល់ផលប្រយោជន៍ដល់បរិស្ថាន — គឺជាកម្មវិធីអនុវត្តជាក់ស្តែងនាពេលបច្ចុប្បន្នដែលកំពុងរីកចម្រើនយ៉ាងឆាប់រហ័ស។
* **ទីផ្សារប្តូរប្រាក់បរទេស (Foreign Exchange - Forex Markets)**
    * *ការផ្តោតលើ (Focus)៖* ទីផ្សារហិរញ្ញវត្ថុធំបំផុតនៅក្នុងពិភពលោកតាមទំហំជួញដូរ ដែលរូបិយប័ណ្ណត្រូវបានទិញ និងលក់ជាបន្តបន្ទាប់។ និស្សិតសិក្សាពីរបៀបដែលអត្រាប្តូរប្រាក់ត្រូវបានកំណត់ អ្វីដែលបណ្តាលឱ្យមានការឡើងថ្លៃ និងការចុះថ្លៃនៃរូបិយប័ណ្ណ និងរបៀបដែលអត្រាប្តូរប្រាក់ដែលប្រែប្រួលខ្លាំងបង្កើតឱ្យមានហានិភ័យសម្រាប់ក្រុមហ៊ុនដែលប្រតិបត្តិការឆ្លងដែនរូបិយប័ណ្ណ exhibition។
* **ទីផ្សារឧបករណ៍ហិរញ្ញវត្ថុនិស្សន្ទ (Derivatives Markets)**
    * *ការផ្តោតលើ (Focus)៖* ឧបករណ៍ហិរញ្ញវត្ថុដែលតម្លៃរបស់វាបានមកពីទ្រព្យសកម្មមូលដ្ឋាន (Underlying Asset) — ដូចជា ជម្រើសទិញលក់ (Options) កិច្ចសន្យាអនាគត (Futures) កិច្ចសន្យាប្តូរដូរ (Swaps) និងកិច្ចសន្យាប្រគល់ទំនិញមុន (Forwards)។ កិច្ចសន្យាអនាគតលើទំនិញកសិកម្ម និងវត្ថុធាតុដើមអនុញ្ញាតឱ្យកសិករកំណត់ថ្លៃសម្រាប់ទិន្នផលដែលមិនទាន់ប្រមូលផល ចំណែកឯកិច្ចសន្យាអនាគតលើប្រេងអនុញ្ញាតឱ្យក្រុមហ៊ុនអាកាសចរណ៍ការពារហានិភ័យថ្លៃដើមប្រេង (Hedge Fuel Costs)។ ឧបករណ៍ហិរញ្ញវត្ថុនិស្សន្ទអាកាសធាតុ (Weather Derivatives) និងឥណទានកាបូន (Carbon Credits) គឺជាកម្មវិធីពាក់ព័ន្ធនឹងនិរន្តរភាព។
* **ទីផ្សារទំនិញកសិកម្ម និងវត្ថុធាតុដើម (Commodity Markets)**
    * *ការផ្តោតលើ (Focus)៖* ទីផ្សារសម្រាប់វត្ថុធាតុដើម — ផលិតផលកសិកម្ម (ស្រូវសាលី សណ្តែកសៀង កាហ្វេ) លោហធាតុ (ទង់ដែង មាស) និងថាមពល (ប្រេង ឧស្ម័នធម្មជាតិ)។ ថ្លៃទំនិញទាំងនេះតភ្ជាប់ខ្សែសង្វាក់ផ្គត់ផ្គង់សកលទៅនឹងជីវភាពរស់នៅរបស់ប្រជាជនក្នុងតំបន់ ហើយមានភាពរសើបខ្លាំងចំពោះព្រឹត្តិការណ៍ភូមិសាស្ត្រនយោបាយ វិបត្តិអាកាសធាតុ និងការប្រែប្រួលរូបិយប័ណ្ណ។

## ២. របៀបដែលផ្សារភាគហ៊ុនដំណើរការ (How Stock Exchanges Work)

* **យន្តការទីផ្សារ (Market Mechanics)**
    * *ការផ្តោតលើ (Focus)៖* របៀបដែលបញ្ជាទិញលក់ត្រូវបានដាក់ ផ្គូផ្គង និងអនុវត្តនៅលើផ្សារភាគហ៊ុនដូចជា NYSE, NASDAQ, ផ្សារភាគហ៊ុនឡុងដ៍, ផ្សារភាគហ៊ុនហុងកុង និងផ្សារភាគហ៊ុនស៊ាងហៃ។ និស្សិតសិក្សាអំពីគម្លាតថ្លៃទិញលក់ (Bid-Ask Spreads) ក្រុមហ៊ុនបង្កើតសាច់ប្រាក់ងាយស្រួលក្នុងទីផ្សារ (Market-Makers) និងតួនាទីនៃប្រព័ន្ធជួញដូរអេឡិចត្រូនិកដែលបច្ចុប្បន្នគ្រប់គ្រងទំហំជួញដូរភាគច្រើនលើសលប់នៅទូទាំងពិភពលោក។
* **សន្ទស្សន៍ភាគហ៊ុនជាសូចនាករសេដ្ឋកិច្ច (Stock Indices as Economic Indicators)**
    * *ការផ្តោតលើ (Focus)៖* សន្ទស្សន៍ឧស្សាហកម្ម Dow Jones, S&P 500, FTSE 100, Nikkei 225 និងសន្ទស្សន៍ Hang Seng ត្រូវបានណែនាំជាឧបករណ៍វាស់ស្ទង់អារម្មណ៍វិនិយោគិន និងសុខភាពសេដ្ឋកិច្ច។ និស្សិតវិភាគពីអ្វីដែលសមាសភាពសន្ទស្សន៍បង្ហាញអំពីឧស្សាហកម្មដែលគ្រប់គ្រងសេដ្ឋកិច្ច និងរបៀបដែលដំណើរការសន្ទស្សន៍អាចដើរយឺតជាង ឬលឿនជាងសកម្មភាពសេដ្ឋកិច្ចជាក់ស្តែង។
* **បទប្បញ្ញត្តិទីផ្សារ និងការត្រួតពិនិត្យ (Market Regulation and Oversight)**
    * *ការផ្តោតលើ (Focus)៖* តួនាទីរបស់ស្ថាប័នគ្រប់គ្រង (ដូចជា SEC នៅសហរដ្ឋអាមេរិក, FCA នៅចក្រភពអង់គ្លេស, SEBI នៅឥណ្ឌា) ក្នុងការរក្សាសុចរិតភាពទីផ្សារ — បង្ការការជួញដូរដោយប្រើប្រាស់ព័ត៌មានផ្ទៃក្នុង (Insider Trading) តម្រូវឱ្យមានការលាតត្រដាងព័ត៌មាន និងការការពារវិនិយោគិនបុគ្គល។ វិបត្តិហិរញ្ញវត្ថុសកលឆ្នាំ ២០០៨ ត្រូវបានសិក្សាជាករណីសិក្សាអំពីបរាជ័យនៃបទប្បញ្ញត្តិ និងហានិភ័យជាប្រព័ន្ធ (Systemic Risks) ដែលកើតឡើងនៅពេលដែលនវានុវត្តន៍ហិរញ្ញវត្ថុដើរលឿនជាងការត្រួតពិនិត្យ។

## ៣. មូលដ្ឋានគ្រឹះនៃពាណិជ្ជកម្មអន្តរជាតិ (International Trade Basics)

* **ការនាំចូល នាំចេញ និងជញ្ជីងពាណិជ្ជកម្ម (Imports, Exports, and the Balance of Trade)**
    * *ការផ្តោតលើ (Focus)៖* ប្រទេសមួយនាំចេញទំនិញ និងសេវាកម្មដែលខ្លួនផលិត និងលក់ទៅឱ្យប្រទេសផ្សេងទៀត ហើយនាំចូលអ្វីដែលខ្លួនទិញពីបរទេស។ ជញ្ជីងពាណិជ្ជកម្ម (ការនាំចេញដកការនាំចូល) គឺជាសូចនាករសេដ្ឋកិច្ចម៉ាក្រូដែលត្រូវបានតាមដានយ៉ាងយកចិត្តទុកដាក់។ និស្សិតពិនិត្យមើលពីរបៀបដែលឱនភាព និងអតិរេកពាណិជ្ជកម្មកើតឡើង និងមូលហេតុដែលវាមានឥទ្ធិពលខាងនយោបាយ។
* **ផលប្រៀបប្រៀបធៀប (Comparative Advantage)**
    * *ការផ្តោតលើ (Focus)៖* អំណះអំណាងជាគ្រឹះរបស់លោក David Ricardo ថា ប្រទេសនានាគួរតែជំនាញលើការផលិតទំនិញណាដែលខ្លួនមានផលប្រៀបប្រៀបធៀប (ថ្លៃដើមធៀប) ហើយធ្វើការជួញដូរសម្រាប់ទំនិញដែលនៅសល់ — ទោះបីជាប្រទេសមួយមានប្រសិទ្ធភាពដាច់ខាតក្នុងការផលិតអ្វីៗគ្រប់យ៉ាងក៏ដោយ។ និស្សិតវាយតម្លៃលើការសន្មតនៃទ្រឹស្តីនេះ និងដែនកំណត់របស់វានៅពេលអនុវត្តចំពោះឧស្សាហកម្មដែលមានផលប៉ះពាល់បរិស្ថានធំធេង។
* **គណនេយ្យជញ្ជីងទូទាត់ (Balance of Payments Accounting)**
    * *ការផ្តោតលើ (Focus)៖* របាយការណ៍គណនេយ្យចុះទ្វេ (Double-Entry Accounting) ដ៏ទូលំទូលាយដែលកត់ត្រារាល់ប្រតិបត្តិការហិរញ្ញវត្ថុរវាងប្រទេសមួយ និងពិភពលោកដែលនៅសល់ — រួមមាន គណនីចរន្ត (Current Account - ទំនិញ សេវាកម្ម ចំណូល ការផ្ទេរប្រាក់) និងគណនីដើមទុន (Capital Account - លំហូរវិនិយោគ)។ និស្សិតរៀនអានទិន្នន័យជញ្ជីងទូទាត់ និងយល់ពីអ្វីដែលវាបង្ហាញអំពីស្ថានភាពសេដ្ឋកិច្ចរបស់ប្រទេសមួយ។
* **ប្រព័ន្ធអត្រាប្តូរប្រាក់ (Exchange Rate Systems)**
    * *ការផ្តោតលើ (Focus)៖* អត្រាប្តូរប្រាក់ថេរ (ដែលរដ្ឋាភិបាលភ្ជាប់រូបិយប័ណ្ណរបស់ខ្លួនទៅនឹងរូបិយប័ណ្ណមួយទៀត) អត្រាប្តូរប្រាក់បណ្តែត (កំណត់ដោយការផ្គត់ផ្គង់ និងតម្រូវការទីផ្សារ) និងប្រព័ន្ធអត្រាប្តូរប្រាក់បណ្តែតមានការគ្រប់គ្រង (Managed Float Systems)។ និស្សិតវិភាគពីរបៀបដែលរបបអត្រាប្តូរប្រាក់ប៉ះពាល់ដល់ភាពប្រកួតប្រជែងពាណិជ្ជកម្ម លំហូរដើមទុន និងភាពបត់បែននៃគោលនយោបាយរូបិយវត្ថុ។

## ៤. ស្ថាប័នអន្តរជាតិ (International Institutions)

* **មូលនិធិហិរញ្ញវត្ថុអន្តរជាតិ (International Monetary Fund - IMF)**
    * *ការផ្តោតលើ (Focus)៖* IMF ត្រូវបានបង្កើតឡើងនៅ Bretton Woods ក្នុងឆ្នាំ ១៩៤៤ ដើម្បីលើកកម្ពស់កិច្ចសហប្រតិបត្តិការរូបិយវត្ថុអន្តរជាតិ និងផ្តល់ជំនួយគាំទ្រជញ្ជីងទូទាត់បន្ទាន់ដល់ប្រទេសជាសមាជិក។ និស្សិតពិនិត្យមើលកម្មវិធីកែសម្រួលរចនាសម្ព័ន្ធរបស់ IMF និងការជជែកដេញដោលអំពីផលវិបាកសេដ្ឋកិច្ច និងសង្គមរបស់ពួកគេ — ជាពិសេសនៅក្នុងប្រទេសកំពុងអភិវឌ្ឍន៍។
* **ធនាគារពិភពលោក (World Bank)**
    * *ការផ្តោតលើ (Focus)៖* ក្រុមធនាគារពិភពលោកផ្តល់ហិរញ្ញប្បទាន សេវាកម្មប្រឹក្សាយោបល់ និងការស្រាវជ្រាវដើម្បីលើកកម្ពស់ការអភិវឌ្ឍន៍នៅក្នុងប្រទេសដែលមានចំណូលទាប និងមធ្យម។ និស្សិតពិនិត្យមើលគម្រោងអភិវឌ្ឍន៍ធំៗ ការវិវត្តនៃគោលនយោបាយការពារបរិស្ថានរបស់ធនាគារពិភពលោក និងតួនាទីរបស់ខ្លួនក្នុងការផ្តល់ហិរញ្ញប្បទានដល់ហេដ្ឋារចនាសម្ព័ន្ធថាមពលកកើតឡើងវិញ។
* **អង្គការពាណិជ្ជកម្មពិភពលោក (World Trade Organization - WTO)**
    * *ការផ្តោតលើ (Focus)៖* ស្ថាប័នអន្តរជាតិដែលកំណត់ និងអនុវត្តច្បាប់នៃពាណិជ្ជកម្មអន្តរជាតិ កាត់សេចក្តីវិវាទពាណិជ្ជកម្ម និងផ្តល់វេទិកាសម្រាប់ការចរចាពាណិជ្ជកម្ម។ និស្សិតវិភាគច្បាប់របស់ WTO លើពន្ធគយ ការឧបត្ថម្ភធន និងរបាំងមិនមែនពន្ធគយ — និងរបៀបដែលច្បាប់ទាំងនោះមានអន្តរកម្មជាមួយគោលនយោបាយបរិស្ថាន (ឧទាហរណ៍ យន្តការកែសម្រួលព្រំដែនកាបូន - Carbon Border Adjustment Mechanisms)។

## ៥. សាជីវកម្មពហុជាតិ និងខ្សែសង្វាក់ផ្គត់ផ្គង់សកល (Multinational Corporations and Global Supply Chains)

* **របៀបដែលក្រុមហ៊ុនពហុជាតិដំណើរការ (How Multinationals Operate)**
    * *ការផ្តោតលើ (Focus)៖* សាជីវកម្មពហុជាតិ (Multinational Corporations - MNCs) រៀបចំការផលិតឆ្លងកាត់ប្រទេសជាច្រើនដើម្បីកាត់បន្ថយថ្លៃដើម ចូលទៅកាន់ទីផ្សារ និងគ្រប់គ្រងបរិយាកាសបទប្បញ្ញត្តិ។ និស្សិតសិក្សាអំពីការវិនិយោគផ្ទាល់ពីបរទេស (Foreign Direct Investment - FDI) ភាពខុសគ្នារវាងក្រុមហ៊ុនបុត្រសម្ព័ន្ធ និងក្រុមហ៊ុនបណ្តាក់ទុនរួមគ្នា (Subsidiaries and Joint Ventures) ការកំណត់ថ្លៃផ្ទេរ (Transfer Pricing) និងយុទ្ធសាស្ត្រធ្វើឱ្យប្រសើរឡើងនូវពន្ធដែលប្រើប្រាស់យុត្តាធិការក្រៅប្រទេស (Offshore Jurisdictions)។
* **ទិដ្ឋភាពទូទៅនៃខ្សែសង្វាក់ផ្គត់ផ្គង់សកល (Global Supply Chain Overview)**
    * *ការផ្តោតលើ (Focus)៖* ផលិតផលទំនើបៗត្រូវបានផ្គុំឡើងពីសមាសភាគដែលប្រភពមកពីប្រទេសរាប់សិប។ និស្សិតតាមដានខ្សែសង្វាក់ផ្គត់ផ្គង់នៃផលិតផលប្រើប្រាស់ទូទៅ — ដូចជា ទូរស័ព្ទវៃឆ្លាត ស្បែកជើងប៉ាតា មួយពែងកាហ្វេ — ចាប់តាំងពីការទាញយកវត្ថុធាតុដើម ការផលិត ឡូជីស្ទីក ការលក់រាយ រហូតដល់ការបោះចោលនៅចុងបញ្ចប់នៃវដ្តជីវិតផលិតផល។ ចំណុចរសើបផ្នែកបរិស្ថាន និងសង្គម (Environmental and Social Hotspots) នៅក្នុងខ្សែសង្វាក់ផ្គត់ផ្គង់ត្រូវបានកំណត់ថាជាគោលដៅអាទិភាពខ្ពស់បំផុតសម្រាប់ការកែលម្អនិរន្តរភាព។
* **គំរូនៃចរន្តវិលជុំសេដ្ឋកិច្ច (Circular Flow Model)**
    * *ការផ្តោតលើ (Focus)៖* គំរូនៃចរន្តវិលជុំបង្ហាញពីរបៀបដែលលុយកាក់ ទំនិញ និងកត្តាផលិតកម្មហូររវាងគ្រួសារ ក្រុមហ៊ុន រដ្ឋាភិបាល និងវិស័យបរទេស។ និស្សិតប្រើប្រាស់គំរូនេះដើម្បីយល់ពីរបៀបដែលវិបត្តិពាណិជ្ជកម្ម ការចំណាយរបស់រដ្ឋាភិបាល និងការសម្រេចចិត្តវិនិយោគរីករាលដាលនៅក្នុងសេដ្ឋកិច្ច។

## ក្របខ័ណ្ឌការងារ និងឧបករណ៍សំខាន់ៗ (Key Frameworks & Tools)

* គំរូនៃចរន្តវិលជុំសេដ្ឋកិច្ច (Circular Flow Model) (ជាមួយវិស័យអន្តរជាតិ)
* គណនេយ្យជញ្ជីងទូទាត់ (Balance of Payments Accounting) (គណនីចរន្ត + គណនីដើមទុន)
* ការវិភាគផលប្រៀបប្រៀបធៀប (Comparative Advantage Analysis)
* ការផ្គត់ផ្គង់ និងតម្រូវការរូបិយប័ណ្ណ (Forex)
* ទំនាក់ទំនងរវាងថ្លៃសញ្ញាបណ្ណ និងអត្រាផលត្រឡប់ (Bond Pricing and Yield Relationship)
* ក្របខ័ណ្ឌការវិនិយោគលើបរិស្ថាន សង្គម និងអភិបាលកិច្ច (ESG Investing Framework)
* ស្តង់ដារសញ្ញាបណ្ណបៃតង (Green Bond Standards) (គោលការណ៍សញ្ញាបណ្ណបៃតង ICMA)
* ការរៀបចំផែនទីខ្សែសង្វាក់ផ្គត់ផ្គង់ (Supply Chain Mapping)

## ហេតុអ្វីបានជាចំណុចនេះសំខាន់សម្រាប់និរន្តរភាពអាជីវកម្ម (Why This Matters for Business Sustainability)

ការផ្លាស់ប្តូរទៅរកសេដ្ឋកិច្ចសកលប្រកបដោយនិរន្តរភាពតម្រូវឱ្យមានការបែងចែកដើមទុនដ៏ធំធេងឡើងវិញ — ចាកចេញឆ្ងាយពីឥន្ធនៈហ្វូស៊ីល និងឧស្សាហកម្មទាញយកធនធាន ឆ្ពោះទៅរកថាមពលកកើតឡើងវិញ កសិកម្មបង្កើតឡើងវិញ (Regenerative Agriculture) និងកម្មន្តសាលវិលជុំ (Circular Manufacturing)។ ដើមទុននោះហូរឆ្លងកាត់ទីផ្សារហិរញ្ញវត្ថុដែលបានសិក្សានៅក្នុងមុខវិជ្ជានេះ។ និស្សិតដែលយល់ពីទីផ្សារសញ្ញាបណ្ណអាចវាយតម្លៃសញ្ញាបណ្ណបៃតង និស្សិតដែលយល់ពីទីផ្សារភាគហ៊ុនអាចវិភាគយុទ្ធសាស្ត្រមូលនិធិ ESG និងផលប៉ះពាល់ជាក់ស្តែងរបស់វាទៅលើឥរិយាបថសាជីវកម្ម ចំណែកឯនិស្សិតដែលយល់ពីទីផ្សារទំនិញកសិកម្ម និងវត្ថុធាតុដើមអាចវាយតម្លៃពីរបៀបដែលសញ្ញាថ្លៃ (ឬការខ្វះខាតរបស់វា) ជម្រុញ ឬរារាំងការផ្លាស់ប្តូរថាមពល។

ខ្សែសង្វាក់ផ្គត់ផ្គង់សកលគឺជាយន្តការចម្បងដែលការសម្រេចចិត្តអាជីវកម្មនៅក្នុងប្រទេសមួយ បង្កើតឱ្យមានលទ្ធផលបរិស្ថាន និងសង្គមនៅក្នុងប្រទេសដទៃទៀត។ ការសម្រេចចិត្តស្វែងរកប្រភពផ្គត់ផ្គង់របស់ម៉ាកសម្លៀកបំពាក់នៅក្នុងទីក្រុងទីស្នាក់ការកណ្តាលរបស់ខ្លួន កំណត់លក្ខខណ្ឌការងារនៅក្នុងរោងចក្រកាត់ដេរនៅក្នុងប្រទេសបង់ក្លាដែស និងការបំពុលទឹកនៅក្នុងទន្លេនានាក្នុងប្រទេសកម្ពុជា។ និស្សិតដែលយល់ពីរបៀបដែលសាជីវកម្មពហុជាតិរៀបចំរចនាសម្ព័ន្ធខ្សែសង្វាក់ផ្គត់ផ្គង់របស់ពួកគេ — និងរបៀបដែលលំហូរដើមទុនផ្តល់ឱ្យទីស្នាក់ការកណ្តាលនូវឥទ្ធិពលលើអ្នកផ្គត់ផ្គង់ — ត្រូវបានបំពាក់ដោយសមត្ថភាពដើម្បីបង្កើតកម្មវិធីនិរន្តរភាពដែលមានន័យ ជាជាងការធ្វើលំហាត់អនុលោមតាមច្បាប់លើផ្ទៃក្រៅ។

ស្ថាប័នអន្តរជាតិដែលបានសិក្សានៅក្នុងមុខវិជ្ជានេះ — IMF, ធនាគារពិភពលោក និង WTO — គឺស្ថិតក្នុងចំណោមតួអង្គសំខាន់បំផុតនៅក្នុងអភិបាលកិច្ចនិរន្តរភាពសកល។ ការប្តេជ្ញាចិត្តផ្នែកហិរញ្ញវត្ថុអាកាសធាតុរបស់ធនាគារពិភពលោក ការកាត់សេចក្តីវិវាទរបស់ WTO អំពីការឧបត្ថម្ភធនលើថាមពលកកើតឡើងវិញ និងការបញ្ចូលហានិភ័យអាកាសធាតុទៅក្នុងការវាយតម្លៃបំណុលរដ្ឋរបស់ IMF សុទ្ធសឹងតែជាទិដ្ឋភាពគោលនយោបាយសកម្ម។ និស្សិតដែលចូលប្រឡូកក្នុងធុរកិច្ចអន្តរជាតិដោយគ្មានចំណេះដឹងផ្នែកស្ថាប័នទាំងនេះ គឺដូចជាការធ្វើដំណើរដោយគ្មានផែនទីដូច្នោះដែរ។

## របៀបដែលមុខវិជ្ជានេះត្រូវបានបង្រៀន (How This Is Taught)

### របៀបបង្រៀន (Teaching Style)

មុខវិជ្ជា សេចក្តីផ្តើមអំពីទីផ្សារហិរញ្ញវត្ថុសកល (Introduction to Global Financial Markets) ត្រូវបានបង្រៀនជាចម្បងតាមរយៈការបង្រៀនផ្អែកលើព្រឹត្តិការណ៍បច្ចុប្បន្ន រួមផ្សំជាមួយនឹងលំហាត់វិភាគទិន្នន័យដែលមានរចនាសម្ព័ន្ធច្បាស់លាស់។ សាស្ត្រាចារ្យចាប់ផ្តើមថ្នាក់នីមួយៗដោយការទាញយកទិន្នន័យទីផ្សារជាក់ស្តែង — ដូចជា ការប្រែប្រួលនៃសន្ទស្សន៍ S&P 500 ប្រចាំថ្ងៃ អត្រាប្តូរប្រាក់ EUR/USD ការឡើងថ្លៃទំនិញកសិកម្ម និងវត្ថុធាតុដើម — ហើយសួរនិស្សិតឱ្យពន្យល់ពីចលនាទាំងនោះដោយប្រើប្រាស់ក្របខ័ណ្ឌការងារដែលបានសិក្សានៅក្នុងសប្តាហ៍នោះ។ ការបង្រៀនផ្លូវការគ្របដណ្តប់លើយន្តការស្ថាប័ន (របៀបដែលការដេញថ្លៃសញ្ញាបណ្ណដំណើរការ របៀបដែលធនាគារកណ្តាលជ្រៀតជ្រែកក្នុងទីផ្សារប្តូរប្រាក់បរទេស របៀបអានទិន្នន័យជញ្ជីងពាណិជ្ជកម្ម)។ វាគ្មិនកិត្តិយសមកពីធនាគារក្នុងតំបន់ ក្រុមហ៊ុនគ្រប់គ្រងការវិនិយោគ ឬស្ថាប័នហិរញ្ញវត្ថុអភិវឌ្ឍន៍ នឹងចូលរួមជាច្រើនដងក្នុងមួយឆមាស ដើម្បីផ្សារភ្ជាប់ទស្សនទានក្នុងថ្នាក់រៀនទៅនឹងការអនុវត្តវិជ្ជាជីវៈជាក់ស្តែង។

### សកម្មភាពក្រុម និងកិច្ចសហការ (Team & Collaborative Activities)
- **គម្រោងតាមដានខ្សែសង្វាក់ផ្គត់ផ្គង់ (Supply Chain Tracing Project)** — ក្រុមដែលមានសមាជិកបីនាក់ជ្រើសរើសផលិតផលប្រើប្រាស់សកលមួយ (ទូរស័ព្ទវៃឆ្លាត ដុំសូកូឡា ស្បែកជើងកីឡាមួយគូ) ហើយរៀបចំផែនទីខ្សែសង្វាក់ផ្គត់ផ្គង់ទាំងមូលចាប់តាំងពីការទាញយកវត្ថុធាតុដើមរហូតដល់ការលក់រាយ។ ក្រុមការងារត្រូវតែកំណត់រូបិយប័ណ្ណដែលប្រើប្រាស់នៅតាមថ្នាំង (Node) នីមួយៗ ឧបករណ៍ហិរញ្ញវត្ថុដែលការពារហានិភ័យសំខាន់ៗ និងចំណុចរសើបផ្នែកបរិស្ថាន ឬសង្គមយ៉ាងហោចណាស់ពីរ។ ការធ្វើបទបង្ហាញរយៈពេល ១០ នាទីចុងក្រោយទៅកាន់ថ្នាក់រៀន រួមបញ្ចូលទាំងផែនទីមើលឃើញ និងម៉ាទ្រីសហានិភ័យជាលាយលក្ខណ៍អក្សរ។
- **ការវិភាគផលប័ត្រសញ្ញាបណ្ណបៃតង (Green Bond Portfolio Analysis)** — ធ្វើការជាគូ និស្សិតទទួលបានផលប័ត្រសិប្បនិម្មិតទំហំ ១០ លានដុល្លារ ហើយត្រូវបែងចែកវាទៅលើសញ្ញាបណ្ណបៃតងពិតប្រាកដយ៉ាងហោចណាស់ចំនួនប្រាំដែលបច្ចុប្បន្នកំពុងចុះបញ្ជីនៅលើទីផ្សារ។ គូនីមួយៗត្រូវតែបញ្ជាក់ពីហេតុផលនៃការបែងចែករបស់ពួកគេដោយប្រើប្រាស់លក្ខណៈវិនិច្ឆ័យនៃគោលការណ៍សញ្ញាបណ្ណបៃតង ICMA ទិន្នន័យអត្រាផលត្រឡប់ និងការវាយតម្លៃលើភាពគួរឱ្យទុកចិត្តផ្នែកបរិស្ថានរបស់អ្នកបោះផ្សាយ។
- **ការពិសោធន៍ពិគ្រោះយោបល់មាត្រា IV របស់ IMF (Mock IMF Article IV Consultation)** — ក្រុមដែលមានសមាជិកប្រាំនាក់ធ្វើការពិសោធន៍កិច្ចប្រជុំត្រួតពិនិត្យរបស់ IMF សម្រាប់ប្រទេសកំពុងអភិវឌ្ឍន៍ដែលបានចាត់តាំង។ ក្រុមរងមួយដើរតួជាបុគ្គលិក IMF ដែលបង្ហាញទិន្នន័យជញ្ជីងទូទាត់របស់ប្រទេស និងកំណត់ហានិភ័យ ចំណែកឯក្រុមរងមួយទៀតដើរតួជាក្រសួងហិរញ្ញវត្ថុរបស់ប្រទេសដែលឆ្លើយតបទៅនឹងអនុសាសន៍។ ការបូកសរុបឡើងវិញផ្តោតលើរបៀបដែលលក្ខខណ្ឌរបស់ IMF មានអន្តរកម្មជាមួយនឹងតម្រូវការហិរញ្ញវត្ថុអាកាសធាតុ។

### ករណីសិក្សាដែលត្រូវបានប្រើប្រាស់ (Case Studies Used)
- **វិបត្តិហិរញ្ញវត្ថុសកលឆ្នាំ ២០០៨ (The 2008 Global Financial Crisis)** — និស្សិតតាមដានខ្សែសង្វាក់នៃការបង្កើតកម្ចីទិញផ្ទះលំដាប់រង (Subprime Mortgage) នៅសហរដ្ឋអាមេរិក ឆ្លងកាត់មូលបត្រធានាដោយកម្ចីទិញផ្ទះ (Mortgage-Backed Securities) កិច្ចសន្យាដោះដូរការខកខានមិនបានសងឥណទាន (Credit Default Swaps) និងការដួលរលំនៃក្រុមហ៊ុន Lehman Brothers រហូតដល់ការបង្កកឥណទានសកល ដោយពិនិត្យមើលថាតើបរាជ័យនៃបទប្បញ្ញត្តិ និងនវានុវត្តន៍ហិរញ្ញវត្ថុមានអន្តរកម្មគ្នាដូចម្តេច និងហានិភ័យជាប្រព័ន្ធអ្វីខ្លះដែលនៅមិនទាន់ដោះស្រាយ។
- **ប្រព័ន្ធជួញដូរការបំភាយឧស្ម័នរបស់សហភាពអឺរ៉ុបក្នុងនាមជាទីផ្សារដើមទុន (The EU Emissions Trading System as a Capital Market)** — និស្សិតវិភាគថ្លៃលិខិតអនុញ្ញាតកាបូនរបស់ EU ETS ឆ្លងកាត់ដំណាក់កាលប្រតិបត្តិការទាំងបីរបស់ខ្លួន ដោយកំណត់ការសម្រេចចិត្តលើបទប្បញ្ញត្តិ (ការបែងចែកលើសកម្រិតក្នុងដំណាក់កាលទី ១, Brexit, កូវីដ-១៩) ដែលបណ្តាលឱ្យមានការដួលរលំថ្លៃ និងការងើបឡើងវិញ ហើយវាយតម្លៃថាប្រវត្តិថ្លៃនេះមានន័យដូចម្តេចចំពោះភាពគួរឱ្យទុកចិត្តរយៈពេលវែងនៃគោលនយោបាយអាកាសធាតុផ្អែកលើទីផ្សារ។
- **វិស័យកាត់ដេររបស់ប្រទេសបង់ក្លាដែស និងហានិភ័យរូបិយប័ណ្ណ (Bangladesh's Garment Sector and Currency Risk)** — និស្សិតពិនិត្យមើលពីរបៀបដែលរោងចក្រកាត់ដេររបស់បង់ក្លាដែស និងអតិថិជនលក់រាយអឺរ៉ុបរបស់ពួកគេគ្រប់គ្រងទំនាក់ទំនងអត្រាប្តូរប្រាក់ BDT/EUR មូលហេតុដែលរោងចក្រទទួលរងហានិភ័យរូបិយប័ណ្ណភាគច្រើន និងរបៀបដែលភាពមិនស៊ីសង្វាក់គ្នានេះប្រមូលផ្តុំសម្ពាធហិរញ្ញវត្ថុទៅលើតំណភ្ជាប់ដែលមានរ៉េបប្រាក់ចំណេញទាបបំផុតនៅក្នុងខ្សែសង្វាក់ផ្គត់ង់។

### ការពិសោធន៍ និងការដើរតួសម្តែង (Simulations & Role-Play)
- **ការពិសោធន៍ជាន់ជួញដូរ Forex (Forex Trading Floor Simulation)** — និស្សិតត្រូវបានចាត់តាំងជាអ្នកទិញ និងអ្នកលក់នៃគូរូបិយប័ណ្ណចំនួនបី (USD/EUR, USD/JPY, USD/BRL) ហើយទទួលបានកាតសេណារីយោដែលមានព័ត៌មានសេដ្ឋកិច្ច (ការតម្លើងអត្រាការប្រាក់គួរឱ្យភ្ញាក់ផ្អើល ការប្រកាសឱនភាពពាណិជ្ជកម្ម)។ ពួកគេចរចាអត្រាប្តូរប្រាក់ភ្លាមៗ រួចជួបជុំគ្នាឡើងវិញដើម្បីសង្កេតមើលថាតើប្រតិបត្តិការសរុបរបស់ថ្នាក់រៀនធ្វើឱ្យអត្រាប្តូរប្រាក់ពិសោធន៍ផ្លាស់ទីដូចម្តេច — ដោយពង្រឹងយន្តការការផ្គត់ផ្គង់ និងតម្រូវការនៃការកំណត់ថ្លៃ Forex។
- **ការដើរតួវិនិយោគិន-អ្នកគ្រប់គ្រងលើការលាតត្រដាងព័ត៌មាន ESG (Investor-Regulator Role-Play on ESG Disclosure)** — ពាក់កណ្តាលនៃថ្នាក់ដើរតួជាវិនិយោគិនស្ថាប័ន (Institutional Investors) ដែលទាមទារឱ្យមានការលាតត្រដាងព័ត៌មាន ESG បន្ថែមទៀតពីក្រុមហ៊ុនពហុជាតិសិប្បនិម្មិតមួយ ហើយពាក់កណ្តាលទៀតដើរតួជាក្រុមទំនាក់ទំនងវិនិយោគិនរបស់ក្រុមហ៊ុនដែលសម្រេចចិត្តលើអ្វីដែលត្រូវលាតត្រដាង។ សេណារីយោនេះរួមបញ្ចូលទាំងសេចក្តីព្រាងរបាយការណ៍និរន្តរភាពដែលស្របតាម ISSB ជាមួយនឹងចន្លោះប្រហោងមួយចំនួន ហើយភាគីនីមួយៗត្រូវតែជជែកវែកញែកពីទស្សនៈស្ថាប័នរបស់ខ្លួនអំពីអ្វីដែលការលាតត្រដាងព័ត៌មានគឺជាសារៈវន្ត អាចធ្វើទៅបាន និងតម្រូវតាមផ្លូវច្បាប់។

### វិធីសាស្ត្រវាយតម្លៃ (Assessment Methods)
- **អត្ថាធិប្បាយទីផ្សារប្រចាំសប្តាហ៍ (Weekly Market Commentary)** (ការបញ្ជូនចំនួនប្រាំដង, ២០% នៃពិន្ទុ) — និស្សិតបញ្ជូនអត្ថាធិប្បាយជាលាយលក្ខណ៍អក្សរកម្រិត ២០០ ពាក្យរៀងរាល់សប្តាហ៍ ដោយផ្សារភ្ជាប់ព្រឹត្តិការណ៍ទីផ្សារហិរញ្ញវត្ថុបច្ចុប្បន្នទៅនឹងក្របខ័ណ្ឌការងារពីវគ្គសិក្សា ដាក់ពិន្ទុលើភាពត្រឹមត្រូវនៃការអនុវត្តក្របខ័ណ្ឌការងារ និងគុណភាពនៃការវិភាគជាជាងទិសដៅនៃការទស្សន៍ទាយ។
- **ការប្រឡងពាក់កណ្តាលឆមាស (Midterm Exam)** — ការប្រឡងរយៈពេល ៩០ នាទី (២៥% នៃពិន្ទុ) រួមមានលំហាត់គណនេយ្យជញ្ជីងទូទាត់ ការគណនាអត្រាផលត្រឡប់សញ្ញាបណ្ណ សំណួរយន្តការ Forex និងអត្ថបទខ្លីមួយអំពីរបៀបដែលស្ថាប័នអន្តរជាតិជាក់លាក់ណាមួយរៀបចំលំហូរដើមទុននិរន្តរភាព។
- **គម្រោងហិរញ្ញវត្ថុខ្សែសង្វាក់ផ្គត់ផ្គង់ (Supply Chain Finance Project)** — គម្រោងតាមដានខ្សែសង្វាក់ផ្គត់ផ្គង់ជាក្រុម (២៥% នៃពិន្ទុ) វាយតម្លៃលើជម្រៅនៃការវិភាគហិរញ្ញវត្ថុ គុណភាពនៃម៉ាទ្រីសហានិភ័យ និងភាពត្រឹមត្រូវនៃការកំណត់ចំណុចរសើបផ្នែកបរិស្ថាន។
- **សារណាស្រាវជ្រាវចុងក្រោយ (Final Research Paper)** — ក្រដាសស្រាវជ្រាវសរសេរផ្ទាល់ខ្លួនកម្រិត ១,៥០០ ពាក្យ (៣០% នៃពិន្ទុ) វិភាគពីផលប៉ះពាល់ទីផ្សារដើមទុននៃនិន្នាការនិរន្តរភាពជាក់លាក់ណាមួយ (ឧទាហរណ៍ ការរីកចម្រើននៃទីផ្សារសញ្ញាបណ្ណបៃតងរដ្ឋ ការវាយតម្លៃថ្លៃឡើងវិញនៃទ្រព្យសកម្មឥន្ធនៈហ្វូស៊ីលដោយវិនិយោគិនស្ថាប័ន ឬតួនាទីរបស់ធនាគារពិភពលោកក្នុងហិរញ្ញប្បទានបន្សាំនឹងអាកាសធាតុ)។

### របៀបសិក្សាដោយខ្លួនឯងលើមុខវិជ្ជានេះជាមួយ AI (How to Self-Learn This with AI)
1. **អាន និងបកស្រាយទិន្នន័យហិរញ្ញវត្ថុជាក់ស្តែង** — សួរ AI ៖ "សូមពន្យល់ខ្ញុំអំពីអ្វីដែលចលនាអត្រាផលត្រឡប់សញ្ញាបណ្ណរតនាគារសហរដ្ឋអាមេរិករយៈពេល ១០ ឆ្នាំនាពេលបច្ចុប្បន្នមានន័យចំពោះថ្លៃដើមខ្ចីប្រាក់របស់សាជីវកម្ម និងសម្រាប់ការបោះផ្សាយសញ្ញាបណ្ណបៃតង។ នាំខ្ញុំឆ្លងកាត់ទំនាក់ទំនងរវាងថ្លៃសញ្ញាបណ្ណ/អត្រាផលត្រឡប់ ហើយផ្តល់ឧទាហរណ៍ជាលេខជាក់ស្តែងមួយ។"
2. **អនុវត្តគណនេយ្យជញ្ជីងទូទាត់** — សួរ AI ៖ "សូមផ្តល់ឱ្យខ្ញុំនូវទិន្នន័យពាណិជ្ជកម្ម និងការវិនិយោគសិប្បនិម្មិតរបស់ប្រទេសមួយសម្រាប់រយៈពេលមួយឆ្នាំ។ ខ្ញុំនឹងចាត់ថ្នាក់ប្រតិបត្តិការនីមួយៗទៅក្នុងគណនីចរន្ត ឬគណនីដើមទុន ហើយគណនាជញ្ជីង។ សូមប្រាប់ខ្ញុំប្រសិនបើខ្ញុំមានកំហុស និងពន្យល់ពីការព្យាបាលគណនេយ្យត្រឹមត្រូវ។"
3. **សាកល្បងការយល់ដឹងរបស់អ្នកអំពីវិបត្តិឆ្នាំ ២០០៨** — សួរ AI ៖ "សូមចោទសួរខ្ញុំអំពីលំដាប់លំដោយនៃព្រឹត្តិការណ៍នៅក្នុងវិបត្តិហិរញ្ញវត្ថុឆ្នាំ ២០០៨។ ចាប់ផ្តើមជាមួយនឹងការបង្កើតកម្ចីទិញផ្ទះលំដាប់រង (Subprime Mortgages) ហើយជម្រុញឱ្យខ្ញុំពន្យល់ពីជំហានបន្ទាប់នីមួយៗនៅក្នុងខ្សែសង្វាក់។ ប្រសិនបើខ្ញុំខកខានចំណុចណាមួយ ឬយល់ខុសពីមូលហេតុ សូមកែតម្រូវខ្ញុំ។"
4. **វិភាគទ្រព្យសកម្មពិតប្រាកដរបស់មូលនិធិ ESG** — សួរ AI ៖ "ខ្ញុំនឹងរៀបរាប់អំពីទ្រព្យសកម្មកំពូលទាំង ១០ របស់មូលនិធិ ESG ពិតប្រាកដមួយ។ ជួយខ្ញុំវាយតម្លៃថាតើទ្រព្យសកម្មទាំងនោះស្របតាមគោលការណ៍ ESG ដែរឬទេ ឬតើមូលនិធិនោះហាក់ដូចជាកំពុងលាងបៃតង (Greenwashing)។ សួរសំណួរបំភ្លឺខ្ញុំអំពីទ្រព្យសកម្មនីមួយៗ ហើយរុញច្រានខ្ញុំឱ្យការពារការវាយតម្លៃរបស់ខ្ញុំ។"
