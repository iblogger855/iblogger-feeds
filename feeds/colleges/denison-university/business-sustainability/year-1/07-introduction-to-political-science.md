# សេចក្តីផ្តើមអំពីវិទ្យាសាស្ត្រនយោបាយ (Introduction to Political Science) (កម្មវិធីសិក្សាផ្នែកសិល្បៈសេរីរបស់សហរដ្ឋអាមេរិក - US Liberal Arts Curriculum)

សេចក្តីផ្តើមអំពីវិទ្យាសាស្ត្រនយោបាយ (Introduction to Political Science) ត្រូវបានសិក្សានៅឆ្នាំទីមួយនៃជំនាញទ្វេផ្នែកពាណិជ្ជកម្មសកល និងនិរន្តរភាព ជាទូទៅនៅក្នុងឆមាសទីពីរ។ មុខវិជ្ជានេះមិនតម្រូវឱ្យមានការយល់ដឹងពីមុនមកផ្នែកទ្រឹស្តីនយោបាយ ឬរដ្ឋាភិបាលនោះទេ។ មុខវិជ្ជានេះមានវត្តមានឡើងដោយសារតែពាណិជ្ជកម្មសកលមិនបានកើតឡើងនៅក្នុងកន្លែងទំនេរស្ថាប័ននោះទេ៖ គ្រប់ទីផ្សារទាំងអស់ត្រូវបានរៀបចំឡើងដោយអំណាចនយោបាយ, គ្រប់ទំនាក់ទំនងពាណិជ្ជកម្មត្រូវបានគ្រប់គ្រងដោយកិច្ចព្រមព្រៀងដែលបានចរចា, និងរាល់គោលដៅនិរន្តរភាពទាំងអស់ត្រូវតែអនុវត្តតាមរយៈដំណើរការនយោបាយ។ និស្សិតដែលយល់ដឹងពីរបៀបដែលរដ្ឋាភិបាលត្រូវបានបង្កើតឡើង, របៀបដែលស្ថាប័នអន្តរជាតិដំណើរការ, និងរបៀបដែលហានិភ័យនយោបាយរៀបចំការសម្រេចចិត្តវិនិយោគ នឹងដំណើរការជាមួយនឹងផែនទីដែលអ្នកជំនាញដែលទទួលបានតែការបណ្តុះបណ្តាលផ្នែកសេដ្ឋកិច្ច ឬធុរកិច្ចសុទ្ធសាធខ្វះខាត។

## ១. ទម្រង់នៃរដ្ឋាភិបាល (Forms of Government)

* **ប្រជាធិបតេយ្យ (Democracy)**
    * *ការផ្តោតលើ (Focus)៖* រដ្ឋាភិបាលប្រជាធិបតេយ្យទទួលបានភាពស្របច្បាប់របស់ពួកគេពីការយល់ព្រមរបស់ប្រជាជន ដែលត្រូវបានបង្ហាញតាមរយៈការបោះឆ្នោត, ស្ថាប័នតំណាង, និងដែនកំណត់រដ្ឋធម្មនុញ្ញលើអំណាច។ និស្សិតរៀនបែងចែករវាងប្រព័ន្ធប្រធានាធិបតី (សហរដ្ឋអាមេរិក, ប្រេស៊ីល, នីហ្សេរីយ៉ា), ប្រព័ន្ធសភា (ចក្រភពអង់គ្លេស, ឥណ្ឌា, អាល្លឺម៉ង់), និងប្រព័ន្ធចម្រុះ, និងវិភាគពីរបៀបដែលការរចនាស្ថាប័នប៉ះពាល់ដល់លទ្ធផលគោលនយោបាយ។ សម្រាប់និរន្តរភាព សំណួរគន្លឹះគឺថាតើស្ថាប័នប្រជាធិបតេយ្យមានសមត្ថភាពក្នុងការប្តេជ្ញាចិត្តបរិស្ថានរយៈពេលវែងដែលរស់រានមានជីវិតឆ្លងកាត់វដ្តនៃការបោះឆ្នោតដែរឬទេ — ដែលជាបញ្ហាប្រឈមមួយដែលច្រើនតែហៅថា "បញ្ហានៃការផ្តោតរយៈពេលខ្លី (Short-Termism Problem)" នៅក្នុងគោលនយោបាយអាកាសធាតុ។
* **ផ្តាច់ការនិយម (Authoritarianism)**
    * *ការផ្តោតលើ (Focus)៖* របបផ្តាច់ការប្រមូលផ្តុំអំណាច, កម្រិតការប្រកួតប្រជែងនយោបាយ, និងរឹតត្បិតសេរីភាពស៊ីវិល និងសេរីភាពសារព័ត៌មាន។ និស្សិតពិនិត្យមើលវិសាលគមចាប់ពីប្រទេសដែលគ្រប់គ្រងដោយគណបក្សតែមួយ (ចិន, វៀតណាម) រហូតដល់របបចម្រុះ (ហុងគ្រី, ទួរគី) ទៅដល់របបផ្តាច់ការបុគ្គលនិយម។ ផ្តាច់ការនិយមមានភាពពាក់ព័ន្ធដោយផ្ទាល់ទៅនឹងពាណិជ្ជកម្មសកល៖ ប្រទេសដែលមានសេដ្ឋកិច្ចផលិតកម្មធំបំផុត និងអ្នកផលិតទំនិញសំខាន់ៗមួយចំនួននៅលើពិភពលោកត្រូវបានគ្រប់គ្រងដោយរបបផ្តាច់ការ ដែលគោលនយោបាយមិនអាចទស្សន៍ទាយបាន, ហានិភ័យនៃអំពើពុករលួយ, និងកំណត់ត្រាសិទ្ធិការងាររបស់ពួកគេចោទជាបញ្ហាប្រឈមច្បាស់លាស់សម្រាប់ក្រុមហ៊ុនដែលមានការប្តេជ្ញាចិត្ត ESG។
* **សហព័ន្ធនិយម និងវិមជ្ឈការ (Federalism and Decentralization)**
    * *ការផ្តោតលើ (Focus)៖* ប្រព័ន្ធសហព័ន្ធ (សហរដ្ឋអាមេរិក, អាល្លឺម៉ង់, ឥណ្ឌា, ប្រេស៊ីល) បែងចែកអំណាចរវាងរដ្ឋាភិបាលជាតិ និងរដ្ឋាភិបាលថ្នាក់ក្រោមជាតិ ដែលបង្កើតឱ្យមានភាពស្មុគស្មាញនៃបទប្បញ្ញត្តិសម្រាប់អាជីវកម្មដែលដំណើរការឆ្លងកាត់រដ្ឋ ឬខេត្ត។ និស្សិតពិនិត្យមើលពីរបៀបដែលសហព័ន្ធនិយមប៉ះពាល់ដល់បទប្បញ្ញត្តិបរិស្ថានជាពិសេស៖ នៅក្នុងសហរដ្ឋអាមេរិក សមត្ថភាពរបស់រដ្ឋកាលីហ្វ័រញ៉ាក្នុងការកំណត់ស្តង់ដារបំភាយឧស្ម័នរថយន្តតឹងរ៉ឹងជាងរដ្ឋាភិបាលសហព័ន្ធបានធ្វើឱ្យវាក្លាយជាគោលការណ៍បទប្បញ្ញត្តិសកលជាក់ស្តែង។ វិមជ្ឈការក៏បង្កើតឱកាសសម្រាប់ការសាកល្បងគោលនយោបាយផងដែរ — ទីក្រុង និងរដ្ឋអាចអនុម័តគោលនយោបាយអាកាសធាតុលឿនជាងរដ្ឋាភិបាលជាតិនៅពេលដែលសកម្មភាពសហព័ន្ធត្រូវបានរារាំង។

## ២. ទ្រឹស្តីទំនាក់ទំនងអន្តរជាតិ (International Relations Theories)

* **តថនិយម (Realism)**
    * *ការផ្តោតលើ (Focus)៖* ទ្រឹស្តីតថនិយម (Realism) ជឿជាក់ថាប្រព័ន្ធអន្តរជាតិមានសភាពអនាធិបតេយ្យ (គ្មានអំណាចកំពូលពីលើរដ្ឋនានានោះទេ), រដ្ឋគឺជាអ្នកដើរតួឯកត្តជនសមហេតុផលដែលផ្តល់អាទិភាពលើសន្តិសុខ និងការរស់រានមានជីវិតផ្ទាល់ខ្លួនរបស់ពួកគេ, និងទំនាក់ទំនងអន្តរជាតិត្រូវបានជម្រុញជាមូលដ្ឋានដោយការប្រកួតប្រជែងអំណាច។ និស្សិតពិនិត្យមើលអំណាចពន្យល់របស់តថនិយមចំពោះបាតុភូតនានាដូចជាការប្រកួតប្រជែងរវាងមហាអំណាច (សហរដ្ឋអាមេរិក-ចិន), ការរត់ប្រណាំងសព្វាវុធ, គាំពារនិយមពាណិជ្ជកម្ម (Trade Protectionism), និងការបរាជ័យនៃកិច្ចព្រមព្រៀងអាកាសធាតុអន្តរជាតិនៅពេលដែលពួកគេផ្ទុយនឹងផលប្រយោជន៍ជាតិដែលបានយល់ឃើញ។ ផលប៉ះពាល់របស់តថនិយមចំពោះអភិបាលកិច្ចនិរន្តរភាពគឺជារឿងគួរឱ្យសោកស្តាយ៖ សកម្មភាពរួមគ្នាលើបញ្ហាធនធានសកលទាមទារការជម្នះតក្កវិជ្ជានៃឥរិយាបថរដ្ឋដែលបម្រើតែផលប្រយោជន៍ផ្ទាល់ខ្លួន។
* **សេរីនិយម (Liberalism)**
    * *ការផ្តោតលើ (Focus)៖* សេរីនិយមអន្តរជាតិ (Liberal Internationalism) ជឿជាក់ថាស្ថាប័នអន្តរជាតិ, ការពឹងផ្អែកគ្នាទៅវិញទៅមកផ្នែកសេដ្ឋកិច្ច, និងអភិបាលកិច្ចប្រជាធិបតេយ្យជួយកាត់បន្ថយជម្លោះ និងអនុញ្ញាតឱ្យមានកិច្ចសហប្រតិបត្តិការ។ និស្សិតពិនិត្យមើលពីរបៀបដែលទ្រឹស្តីសេរីនិយមបានទ្រទ្រង់សណ្តាប់ធ្នាប់អន្តរជាតិក្រោយឆ្នាំ ១៩៤៥ — អង្គការសហប្រជាជាតិ, IMF, ធនាគារពិភពលោក, GATT/WTO — និងរបៀបដែលសណ្តាប់ធ្នាប់នោះបានអនុញ្ញាតឱ្យមានការពង្រីកពាណិជ្ជកម្មសកល និងការអភិវឌ្ឍជាលំដាប់នៃអភិបាលកិច្ចបរិស្ថានពហុភាគី។ កិច្ចព្រមព្រៀងទីក្រុងប៉ារីស, អនុសញ្ញាស្តីពីជីវចម្រុះ, និងអនុសញ្ញាបាសែល (Basel Convention) ស្តីពីកាកសំណល់គ្រោះថ្នាក់ ត្រូវបានពិនិត្យជាភស្តុតាងដែលថាកិច្ចសហប្រតិបត្តិការអន្តរជាតិលើបញ្ហាបរិស្ថានគឺអាចធ្វើទៅបានក្រោមលក្ខខណ្ឌស្ថាប័នត្រឹមត្រូវ។
* **ស្ថាបនានិយម (Constructivism)**
    * *ការផ្តោតលើ (Focus)៖* ស្ថាបនានិយម (Constructivism) ជជែកថាទំនាក់ទំនងអន្តរជាតិត្រូវបានរៀបចំឡើងមិនត្រឹមតែដោយផលប្រយោជន៍រូបវន្ត និងអំណាចប៉ុណ្ណោះទេ ប៉ុន្តែដោយគំនិត, បទដ្ឋាន, និងអត្តសញ្ញាណ — ហើយចំណុចទាំងនេះអាចផ្លាស់ប្តូរតាមពេលវេលាតាមរយៈការបញ្ចុះបញ្ចូល និងការរៀនសូត្រសង្គម។ និស្សិតពិនិត្យមើលពីរបៀបដែលបទដ្ឋានបរិស្ថានបានវិវត្តជាអន្តរជាតិ៖ ការថយចុះស្រទាប់អូហ្សូនដំបូងឡើយត្រូវបានបដិសេធថាជាបញ្ហាបន្ទាប់បន្សំ; ប៉ុន្តែនៅឆ្នាំ ១៩៨៧ វាបានបង្កើតជាពិធីសារម៉ុងរ៉េអាល់ (Montreal Protocol) ដែលជាកិច្ចព្រមព្រៀងបរិស្ថានអន្តរជាតិដែលទទួលបានជោគជ័យបំផុតមួយនៅក្នុងប្រវត្តិសាស្ត្រ។ ការយល់ឃើញរបស់ស្ថាបនានិយម — ថាអ្វីដែលប្រទេសនានាចង់បាន និងអ្វីដែលពួកគេចាត់ទុកថាស្របច្បាប់អាចផ្លាស់ប្តូរបាន — គឺចាំបាច់ណាស់សម្រាប់យល់ដឹងពីរបៀបដែលបទដ្ឋាននិរន្តរភាពក្លាយជាស្តង់ដារអន្តរជាតិ។

## ៣. អង្គការអន្តរជាតិ (International Organizations)

* **ប្រព័ន្ធអង្គការសហប្រជាជាតិ (The United Nations System)**
    * *ការផ្តោតលើ (Focus)៖* អង្គការសហប្រជាជាតិរួមបញ្ចូលទីភ្នាក់ងារ, កម្មវិធី, និងស្ថាប័នឯកទេសជាង ៣០។ និស្សិតផ្តោតលើស្ថាប័នដែលពាក់ព័ន្ធនឹងពាណិជ្ជកម្មសកល និងនិរន្តរភាពបំផុត៖ កម្មវិធីបរិស្ថានសហប្រជាជាតិ (UNEP), អនុសញ្ញាក្របខ័ណ្ឌសហប្រជាជាតិស្តីពីការប្រែប្រួលអាកាសធាតុ (UNFCCC ដែលរៀបចំការចរចា COP ប្រចាំឆ្នាំ), អង្គការស្បៀងអាហារ និងកសិកម្ម (FAO), និងអង្គការពលកម្មអន្តរជាតិ (ILO)។ គោលដៅអភិវឌ្ឍន៍ប្រកបដោយចីរភាពរបស់អង្គការសហប្រជាជាតិ (UN SDGs) — គោលដៅចំនួន ១៧ ជាមួយនឹងគោលដៅរងចំនួន ១៦៩ ដែលបានអនុម័តក្នុងឆ្នាំ ២០១៥ — ត្រូវបានពិនិត្យជាក្របខ័ណ្ឌការងារដែលក្រុមហ៊ុនពហុជាតិ និងវិនិយោគិនស្ថាប័នឥឡូវនេះប្រើប្រាស់ជាឯកសារយោងសម្រាប់របាយការណ៍ និងការកំណត់គោលដៅ។
* **អង្គការពាណិជ្ជកម្មពិភពលោក (World Trade Organization - WTO)**
    * *ការផ្តោតលើ (Focus)៖* WTO គ្រប់គ្រងច្បាប់ពាណិជ្ជកម្មអន្តរជាតិសម្រាប់ប្រទេសជាសមាជិកចំនួន ១៦៤, កាត់សេចក្តីវិវាទពាណិជ្ជកម្មតាមរយៈស្ថាប័នដោះស្រាយវិវាទរបស់ខ្លួន, និងផ្តល់វេទិកាសម្រាប់ការចរចាពាណិជ្ជកម្មពហុភាគី។ និស្សិតពិនិត្យមើលពីរបៀបដែលច្បាប់របស់ WTO មានអន្តរកម្មជាមួយគោលនយោបាយបរិស្ថាន៖ តើប្រទេសមួយអាចអនុវត្តយន្តការកែសម្រួលព្រំដែនកាបូនលើទំនិញនាំចូលពីប្រទេសដែលគ្មានការកំណត់ថ្លៃកាបូនបានដែរឬទេ? តើវាអាចហាមឃាត់ការនាំចូលទំនិញដែលផលិតដោយពលកម្មបង្ខំបានដែរឬទេ? ការចាត់ចែងរបស់ WTO លើសំណួរទាំងនេះ — និងភាពតានតឹងរវាងសេរីភាវូបនីយកម្មពាណិជ្ជកម្ម និងស្តង់ដារបរិស្ថានព្រមទាំងពលកម្ម — គឺជាព្រំដែនគោលនយោបាយសកម្ម និងមានឥទ្ធិពល។
* **IMF និងធនាគារពិភពលោក (The IMF and World Bank)**
    * *ការផ្តោតលើ (Focus)៖* និស្សិតពង្រីកចំណេះដឹងដែលបានណែនាំក្នុងឆមាសទីមួយអំពីស្ថាប័នទាំងនេះ ដោយពិនិត្យមើលតួនាទីរបស់ពួកគេក្នុងនិរន្តរភាពជាពិសេស៖ ការរួមបញ្ចូលហានិភ័យអាកាសធាតុរបស់ IMF ទៅក្នុងការវាយតម្លៃបំណុលរដ្ឋ និងការវាយតម្លៃគោលនយោបាយការប្រែប្រួលអាកាសធាតុរបស់ខ្លួន; ការប្តេជ្ញាចិត្តរបស់ធនាគារពិភពលោកក្នុងការបញ្ចប់ហិរញ្ញប្បទានសម្រាប់ប្រេង និងឧស្ម័នខ្សែទឹកលើ និងតួនាទីរបស់ខ្លួនជាអ្នកផ្តល់ហិរញ្ញវត្ថុអាកាសធាតុពហុភាគីធំបំផុតដល់ប្រទេសកំពុងអភិវឌ្ឍន៍។ រចនាសម្ព័ន្ធអភិបាលកិច្ចនៃស្ថាប័នទាំងពីរ — ទម្ងន់សម្លេងឆ្នោតផ្អែកលើទំហំសេដ្ឋកិច្ច ដែលផ្តល់ឱ្យសហរដ្ឋអាមេរិក និងសហភាពអឺរ៉ុបនូវឥទ្ធិពលមិនសមាមាត្រ — ត្រូវបានពិនិត្យជាប្រភពនៃភាពតានតឹងនយោបាយជាមួយតំបន់សកលខាងត្បូង។

## ៤. ហានិភ័យនយោបាយសម្រាប់អាជីវកម្មពហុជាតិ (Political Risk for Multinational Businesses)

* **អ្វីទៅជាហានិភ័យនយោបាយ (What Political Risk Is)**
    * *ការផ្តោតលើ (Focus)៖* ហានិភ័យនយោបាយសំដៅលើលទ្ធភាពដែលសកម្មភាពរបស់រដ្ឋាភិបាល — ការផ្លាស់ប្តូរគោលនយោបាយ, ការផ្លាស់ប្តូរបទប្បញ្ញត្តិ, ការរឹបអូសទ្រព្យសម្បត្តិ, ចលាចលស៊ីវិល, ឬជម្លោះប្រដាប់អាវុធ — នឹងជះឥទ្ធិពលអវិជ្ជមានដល់ប្រតិបត្តិការ ឬតម្លៃទ្រព្យសកម្មរបស់អាជីវកម្ម។ និស្សិតរៀនពីចំណាត់ថ្នាក់នៃហានិភ័យនយោបាយ៖ ហានិភ័យម៉ាក្រូ (ប៉ះពាល់ដល់គ្រប់ក្រុមហ៊ុនក្នុងប្រទេស), ហានិភ័យមីក្រូ (ប៉ះពាល់ដល់ក្រុមហ៊ុន ឬវិស័យជាក់លាក់), និងហានិភ័យចៃដន្យ (ជម្រុញដោយព្រឹត្តិការណ៍ជាក់លាក់ដូចជាការបោះឆ្នោត ឬរដ្ឋប្រហារ)។ ហានិភ័យនយោបាយត្រូវបានកំណត់ថ្លៃជាប្រចាំទៅក្នុងការសម្រេចចិត្តវិនិយោគតាមរយៈអត្រាបញ្ចុះតម្លៃ, បុព្វលាភធានារ៉ាប់រង, និងយុទ្ធសាស្ត្រចូលទីផ្សារ។
* **ហានិភ័យអធិបតេយ្យភាព និងការរឹបអូសទ្រព្យសម្បត្តិ (Sovereign Risk and Expropriation)**
    * *ការផ្តោតលើ (Focus)៖* ហានិភ័យអធិបតេយ្យភាពគឺជាហានិភ័យដែលរដ្ឋាភិបាលនឹងខកខានមិនបានសងបំណុលរបស់ខ្លួន ឬបដិសេធកាតព្វកិច្ចកិច្ចសន្យារបស់ខ្លួនចំពោះវិនិយោគិនបរទេស។ ការរឹបអូសទ្រព្យសម្បត្តិ (Expropriation) — ការរឹបអូសទ្រព្យសកម្មកម្មសិទ្ធិបរទេសដោយរដ្ឋាភិបាល — គឺជាទម្រង់ហានិភ័យនយោបាយធ្ងន់ធ្ងរបំផុត ហើយត្រូវបានប្រើប្រាស់ជាប្រវត្តិសាស្ត្រនៅក្នុងវិស័យប្រេង និងឧស្ម័ន, វិស័យធ្វើរ៉ែ, និងវិស័យទូរគមនាគមន៍។ និស្សិតពិនិត្យមើលក្របខ័ណ្ឌច្បាប់ (សន្ធិសញ្ញាវិនិយោគទ្វេភាគី, ការអាជ្ញាកណ្តាល ICSID) ដែលវិនិយោគិនប្រើប្រាស់ដើម្បីកាត់បន្ថយហានិភ័យនៃការរឹបអូស និងលក្ខខណ្ឌដែលការរឹបអូសអាចកើតមានឡើងខ្លាំងបំផុត — ជាតិនិយមធនធាន, វិបត្តិសេដ្ឋកិច្ច, ការផ្លាស់ប្តូររបបនយោបាយ។
* **ហានិភ័យបទប្បញ្ញត្តិ និងគោលនយោបាយ (Regulatory and Policy Risk)**
    * *ការផ្តោតលើ (Focus)៖* សូម្បីតែនៅក្នុងប្រទេសប្រជាធិបតេយ្យដែលមានស្ថិរភាព ការផ្លាស់ប្តូរគោលនយោបាយអាចកែប្រែបរិយាកាសធុរកិច្ចយ៉ាងខ្លាំង។ និស្សិតពិនិត្យមើលពីរបៀបដែលការបោះឆ្នោត, ការសម្រេចរបស់តុលាការ, និងសកម្មភាពរបស់ទីភ្នាក់ងារបទប្បញ្ញត្តិបង្កើតឱ្យមានភាពមិនច្បាស់លាស់សម្រាប់ការវិនិយោគដែលមានរយៈពេលវែង — ជាពិសេសនៅក្នុងវិស័យថាមពល និងហេដ្ឋារចនាសម្ព័ន្ធ ដែលទ្រព្យសកម្មមានអាយុកាល ២០-៤០ ឆ្នាំ។ ហានិភ័យនៃ "ទ្រព្យសកម្មជាប់គាំង (Stranded Assets)" — ហេដ្ឋារចនាសម្ព័ន្ធថាមពលផូស៊ីលដែលលែងមានតម្លៃសេដ្ឋកិច្ចដោយសារបទប្បញ្ញត្តិអាកាសធាតុ — គឺជាទម្រង់នៃហានិភ័យបទប្បញ្ញត្តិដែលមានឥទ្ធិពលបំផុតនៅក្នុងបរិបទនិរន្តរភាព ដោយមានទ្រព្យសកម្មប៉ាន់ស្មានប្រហែល ១-៤ ទ្រីលីនដុល្លារដែលអាចរងផលប៉ះពាល់។

## ៥. ភូមិសាស្ត្រនយោបាយ និងពាណិជ្ជកម្ម (Geopolitics and Trade)

* **ទណ្ឌកម្ម និងការត្រួតពិនិត្យការនាំចេញ (Sanctions and Export Controls)**
    * *ការផ្តោតលើ (Focus)៖* រដ្ឋាភិបាលប្រើប្រាស់ទណ្ឌកម្មសេដ្ឋកិច្ច — ការរឹតត្បិតលើពាណិជ្ជកម្ម, ហិរញ្ញវត្ថុ, និងការវិនិយោគ — ដើម្បីដាក់សម្ពាធលើរដ្ឋាភិបាលបរទេស និងតួអង្គមិនមែនរដ្ឋ។ និស្សិតពិនិត្យមើលយន្តការ និងប្រសិទ្ធភាពនៃរបបទណ្ឌកម្ម (អ៊ីរ៉ង់, រុស្ស៊ី, កូរ៉េខាងជើង) និងរបៀបដែលក្រុមហ៊ុនត្រូវតែគ្រប់គ្រងវា៖ បរាជ័យនៃការអនុលោមភាពនាំមកនូវការពិន័យព្រហ្មទណ្ឌ និងរដ្ឋប្បវេណីធ្ងន់ធ្ងរ។ ការត្រួតពិនិត្យការនាំចេញរឹតត្បិតការផ្ទេរបច្ចេកវិទ្យា និងសម្ភារៈរសើបទៅកាន់ប្រទេសដែលបានកំណត់។ ឧបករណ៍ទាំងពីរនេះបានក្លាយជាចំណុចលេចធ្លោនៅពេលដែលការប្រកួតប្រជែងភូមិសាស្ត្រនយោបាយរវាងសហរដ្ឋអាមេរិក និងចិនកាន់តែតានតឹង ដែលបង្កើតឱ្យមានការបែកបាក់នៃខ្សែសង្វាក់ផ្គត់ផ្គង់បច្ចេកវិទ្យាសកលដែលប៉ះពាល់ផ្ទាល់ដល់ក្រុមហ៊ុនក្នុងវិស័យ semiconductor, ថាមពលស្អាត, និងកម្មន្តសាលកម្រិតខ្ពស់។
* **ពន្ធគយ និងគោលនយោបាយពាណិជ្ជកម្ម (Tariffs and Trade Policy)**
    * *ការផ្តោតលើ (Focus)៖* ពន្ធគយគឺជាពន្ធលើទំនិញនាំចូលដែលបង្កើនថ្លៃទំនិញបរទេស ការពារអ្នកផលិតក្នុងស្រុកដោយកាត់ទឹកចិត្តអ្នកប្រើប្រាស់ក្នុងស្រុក។ និស្សិតពិនិត្យមើលសេដ្ឋកិច្ចនយោបាយនៃការការពារពាណិជ្ជកម្ម — ហេតុអ្វីបានជាឧស្សាហកម្មស្វែងរកពន្ធគយ, របៀបដែលរដ្ឋាភិបាលរក្សាតុល្យភាពរវាងផលប្រយោជន៍អ្នកផលិត និងអ្នកប្រើប្រាស់, និងរបៀបដែលយន្តការដោះស្រាយវិវាទរបស់ WTO ដោះស្រាយសកម្មភាពពន្ធគយឯកតោភាគី។ សង្គ្រាមពាណិជ្ជកម្មសហរដ្ឋអាមេរិក-ចិន, យន្តការកែសម្រួលព្រំដែនកាបូនរបស់សហភាពអឺរ៉ុប, និងការពិភាក្សាអំពីភាពស៊ីគ្នានៃគោលនយោបាយឧស្សាហកម្ម (ច្បាប់កាត់បន្ថយអតិផរណា - Inflation Reduction Act, កិច្ចព្រមព្រៀងបៃតងសហភាពអឺរ៉ុប - EU Green Deal) ជាមួយនឹងច្បាប់របស់ WTO ត្រូវបានពិនិត្យជាករណីសិក្សាគោលនយោបាយជាក់ស្តែង។
* **ការបោះឆ្នោត និងអស្ថិរភាពនយោបាយ (Elections and Political Instability)**
    * *ការផ្តោតលើ (Focus)៖* ការបោះឆ្នោតបង្កើតឱ្យមានភាពមិនច្បាស់លាស់នៃគោលនយោបាយ៖ ការផ្លាស់ប្តូររដ្ឋាភិបាលអាចផ្លាស់ប្តូរការប្តេជ្ញាចិត្តបទប្បញ្ញត្តិ, កែប្រែរបបឧបត្ថម្ភធន, និងចរចាកិច្ចព្រមព្រៀងអន្តរជាតិឡើងវិញ។ និស្សិតពិនិត្យមើលពីរបៀបដែលទីផ្សារភាគហ៊ុន និងបំណុលកំណត់ថ្លៃហានិភ័យនៃការបោះឆ្នោត និងរបៀបដែលអាជីវកម្មកសាងការរៀបចំផែនការសេណារីយ៉ូនយោបាយទៅក្នុងការសម្រេចចិត្តវិនិយោគ។ អស្ថិរភាពនយោបាយ — ទោះជាមកពីការបោះឆ្នោតដែលមានជម្លោះ, ការព្យាយាមធ្វើរដ្ឋប្រហារ, ឬចលាចលស៊ីវិលអូសបន្លាយពេល — រំខានដល់ខ្សែសង្វាក់ផ្គត់ផ្គង់, កាត់បន្ថយការវិនិយោគផ្ទាល់ពីបរទេស, និងអាចបង្កឱ្យមានការផ្លាស់ប្តូរភ្លាមៗនៃបទប្បញ្ញត្តិការងារ និងបរិស្ថានដែលប៉ះពាល់ដល់ក្រុមហ៊ុនដែលប្រតិបត្តិការនៅក្នុងប្រទេសរងគ្រោះ។

## ក្របខ័ណ្ឌការងារ និងឧបករណ៍សំខាន់ៗ (Key Frameworks & Tools)

* ការវិភាគរដ្ឋាភិបាលប្រៀបធៀប (ប្រព័ន្ធប្រធានាធិបតី ធៀបនឹង ប្រព័ន្ធសភា - Comparative Government Analysis (presidential vs. parliamentary systems))
* ទ្រឹស្តីទំនាក់ទំនងអន្តរជាតិ (តថនិយម, សេរីនិយម, ស្ថាបនានិយម - International Relations Theory (realism, liberalism, constructivism))
* ចំណាត់ថ្នាក់ហានិភ័យនយោបាយ (ម៉ាក្រូ, មីក្រូ, ចៃដន្យ - Political Risk Taxonomy (macro, micro, contingent))
* ក្របខ័ណ្ឌ UN SDG (គោលដៅ ១៧, គោលដៅរង ១៦៩ - UN SDG Framework (17 Goals, 169 Targets))
* យន្តការដោះស្រាយវិវាទរបស់ WTO (WTO Dispute Settlement Mechanism)

* ក្របខ័ណ្ឌអនុលោមភាពទណ្ឌកម្ម និងការត្រួតពិនិត្យការនាំចេញ (Sanctions and Export Control Compliance Framework)
* ការរៀបចំផែនការសេណារីយ៉ូនយោបាយសម្រាប់ការសម្រេចចិត្តវិនិយោគ (Political Scenario Planning for Investment Decisions)

## ហេតុអ្វីបានជាចំណុចនេះសំខាន់សម្រាប់និរន្តរភាពអាជីវកម្ម (Why This Matters for Business Sustainability)

អស្ថិរភាពនយោបាយគឺជាហានិភ័យមួយក្នុងចំណោមហានិភ័យដែលត្រូវបានគេវាយតម្លៃទាបបំផុតនៅក្នុងខ្សែសង្វាក់ផ្គត់ផ្គង់និរន្តរភាព។ សារធាតុកូបាល់ដែលផ្តល់ថាមពលដល់អាគុយរថយន្តអគ្គិសនីត្រូវបានប្រភពមកពីសាធារណរដ្ឋប្រជាធិបតេយ្យកុងហ្គោជាចម្បង — ដែលជាប្រទេសមានប្រវត្តិនៃជម្លោះប្រដាប់អាវុធ, អំពើពុករលួយរបស់រដ្ឋាភិបាល, និងទីតាំងធ្វើរ៉ែខ្នាតតូចដែលមានការរំលោភបំពានពលកម្មកុមារ និងបរិស្ថានធ្ងន់ធ្ងរ។ ប្រេងដូងនៅក្នុងខ្សែសង្វាក់ផ្គត់ផ្គង់ទំនិញប្រើប្រាស់ត្រូវបានដាំដុះនៅក្នុងប្រទេសឥណ្ឌូនេស៊ី និងម៉ាឡេស៊ី — ប្រទេសដែលច្បាប់កម្មសិទ្ធិដីធ្លី, ការការពារសិទ្ធិជនជាតិដើមភាគតិច, និងការអនុវត្តច្បាប់កាប់បំផ្លាញព្រៃឈើត្រូវបានរៀបចំឡើងយ៉ាងស៊ីជម្រៅដោយឌីណាមិកនយោបាយដែលផ្លាស់ប្តូរទៅតាមការបោះឆ្នោត និងការតែងតាំងមន្ត្រីបទប្បញ្ញត្តិ។ ការយល់ដឹងពីវិទ្យាសាស្ត្រនយោបាយមិនមែនជាចំណេះដឹងផ្ទៃខាងក្រោយសម្រាប់អ្នកជំនាញផ្នែកនិរន្តរភាពនោះទេ — វាគឺជាព័ត៌មានប្រតិបត្តិការផ្ទាល់។

គោលនយោបាយពាណិជ្ជកម្មរៀបចំទិដ្ឋភាពប្រកួតប្រជែងសម្រាប់រាល់ការផ្លាស់ប្តូរនិរន្តរភាព។ តម្រូវការខ្លឹមសារក្នុងស្រុករបស់ច្បាប់កាត់បន្ថយអតិផរណាសហរដ្ឋអាមេរិកសម្រាប់ឥណទានពន្ធថាមពលបៃតង, យន្តការកែសម្រួលព្រំដែនកាបូនរបស់សហភាពអឺរ៉ុប, និងការគ្រប់គ្រងរបស់ប្រទេសចិនលើការផលិតផ្ទាំងសូឡា និងអាគុយ សុទ្ធសឹងតែជាតថភាពពាណិជ្ជកម្មដែលរៀបចំឡើងដោយភូមិសាស្ត្រនយោបាយ។ និស្សិតដែលយល់ដឹងពីរបៀបដែលគោលនយោបាយពាណិជ្ជកម្មត្រូវបានបង្កើត, តស៊ូ, និងអនុវត្ត អាចប៉ាន់ស្មានទុកជាមុនពីរបៀបដែលឌីណាមិកទាំងនេះនឹងវិវត្ត — និងអាចប្រឹក្សាយោបល់ដល់ស្ថាប័នរបស់ពួកគេលើផ្នែកប្រភពផ្គត់ផ្គង់, ការវិនិយោគ, និងយុទ្ធសាស្ត្រទីផ្សារទៅតាមនោះ។

ស្ថាប័នអន្តរជាតិកំណត់កម្រិតទាបបំផុតសម្រាប់អភិបាលកិច្ចនិរន្តរភាពសកល។ ដំណើរការ COP របស់ UNFCCC បង្កើតកិច្ចព្រមព្រៀងដែលរដ្ឋាភិបាលជាតិបកប្រែទៅជាការកំណត់ថ្លៃកាបូន, គោលនយោបាយកាប់បំផ្លាញព្រៃឈើ, និងការប្តេជ្ញាចិត្តជីវចម្រុះ។ ច្បាប់របស់ WTO កំណត់ថាតើរបាំងពាណិជ្ជកម្មដែលជម្រុញដោយអាកាសធាតុអាចការពារបានតាមផ្លូវច្បាប់ដែរឬទេ។ អនុសញ្ញារបស់ ILO កំណត់ស្តង់ដារការងារដែលកម្មវិធីខ្សែសង្វាក់ផ្គត់ផ្គង់និរន្តរភាពយកជាឯកសារយោង។ និស្សិតដែលយល់ដឹងពីរបៀបដែលស្ថាប័នទាំងនេះដំណើរការ — អ្នកណាគ្រប់គ្រងរបៀបវារៈរបស់ពួកគេ, របៀបដែលការសម្រេចចិត្តត្រូវបានធ្វើឡើង, កន្លែងណាដែលជាខ្សែបន្ទាត់ជម្លោះ — នឹងយល់ដឹងពីសេដ្ឋកិច្ចនយោបាយនៃការផ្លាស់ប្តូរនិរន្តរភាព មិនមែនគ្រាន់តែវិទ្យាសាស្ត្របរិស្ថានរបស់វានោះទេ។

## របៀបដែលមុខវិជ្ជានេះត្រូវបានបង្រៀន (How This Is Taught)

### របៀបបង្រៀន (Teaching Style)
សេចក្តីផ្តើមអំពីវិទ្យាសាស្ត្រនយោបាយ ត្រូវបានបង្រៀនតាមរយៈការរួមបញ្ចូលគ្នានៃការបង្រៀនទ្រឹស្តី, ការពិភាក្សាបែបសូក្រាត, និងការវិភាគព្រឹត្តិការណ៍បច្ចុប្បន្នដែលមានរចនាសម្ព័ន្ធច្បាស់លាស់។ សាស្ត្រាចារ្យប្រើប្រាស់ទម្រង់បង្រៀនទ្រឹស្តីដើម្បីណែនាំក្របខ័ណ្ឌការងារ (តថនិយម, សេរីនិយម, ស្ថាបនានិយម) ហើយបន្ទាប់មកសាកល្បងក្របខ័ណ្ឌទាំងនោះភ្លាមៗធៀបនឹងព្រឹត្តិការណ៍ភូមិសាស្ត្រនយោបាយជាក់ស្តែងដែលដកស្រង់ចេញពីព័ត៌មានប្រចាំសប្តាហ៍នោះ — វិវាទពាណិជ្ជកម្ម, លទ្ធផលបោះឆ្នោត, ការប្រកាសទណ្ឌកម្ម។ និស្សិតត្រូវបានរំពឹងទុកថានឹងអានប្រភពព័ត៌មានសំខាន់មួយ (Financial Times, The Economist, ឬ Foreign Affairs) ជារៀងរាល់សប្តាហ៍ និងចូលរួមថ្នាក់រៀនដោយត្រៀមខ្លួនរួចជាស្រេចដើម្បីភ្ជាប់ព្រឹត្តិការណ៍បច្ចុប្បន្នទៅនឹងទ្រឹស្តីវគ្គសិក្សា។ វាគ្មិនកិត្តិយសមកពីវិស័យធុរកិច្ច — អ្នកវិភាគហានិភ័យនយោបាយនៅក្រុមហ៊ុនពហុជាតិ, មេធាវីអនុលោមភាពពាណិជ្ជកម្ម, មន្ត្រីហិរញ្ញវត្ថុអភិវឌ្ឍន៍ — ផ្តល់នូវទស្សនៈអ្នកអនុវត្តជាក់ស្តែងអំពីរបៀបដែលការវិភាគនយោបាយបកប្រែទៅជាការសម្រេចចិត្តវិជ្ជាជីវៈ។

### សកម្មភាពក្រុម និងកិច្ចសហការ (Team & Collaborative Activities)
- **ការពិសោធន៍ការជជែកដេញដោលក្រុមប្រឹក្សាសន្តិសុខអន្តរជាតិ (Mock UN Security Council Debate)** — ក្រុមដែលមានសមាជិកប្រាំនាក់តំណាងឱ្យសមាជិកអចិន្ត្រៃយ៍ ឬសមាជិកប្តូរវេននៃក្រុមប្រឹក្សាសន្តិសុខលើវិបត្តិកម្រិតភូមិសាស្ត្រនយោបាយសន្មតដែលពាក់ព័ន្ធនឹងប្រតិបត្តិការរបស់ក្រុមហ៊ុនពហុជាតិ (ឧទាហរណ៍ ទ្រព្យសកម្មរបស់ក្រុមហ៊ុនធ្វើរ៉ែដែលជាប់នៅក្នុងតំបន់ជម្លោះស៊ីវិល)។ ក្រុមការងារត្រូវតែចរចាតាក់តែងសេចក្តីសម្រេចចិត្តដែលដោះស្រាយវិបត្តិនយោបាយ, សំណួរការទទួលខុសត្រូវរបស់សាជីវកម្ម, និងវិមាត្រមនុស្សធម៌ — ក្រុមនីមួយៗតស៊ូមតិសម្រាប់ផលប្រយោជន៍ប្រទេសដែលបានចាត់តាំង ខណៈពេលដែលព្យាយាមកសាងសម្ព័ន្ធភាពដែលចាំបាច់សម្រាប់ការអនុម័ត។
- **តារាងពិន្ទុហានិភ័យភូមិសាស្ត្រនយោបាយ (Geopolitical Risk Scorecard)** — ក្រុមដែលមានសមាជិកបីនាក់ត្រូវបានចាត់តាំងឱ្យសិក្សាលើប្រទេសពិតប្រាកដមួយដែលដើរតួជាម្ចាស់ផ្ទះនៃសកម្មភាពខ្សែសង្វាក់ផ្គត់ផ្គង់សកលសំខាន់ៗ (ឧទាហរណ៍ DRC សម្រាប់កូបាល់, វៀតណាមសម្រាប់គ្រឿងអេឡិចត្រូនិក, ឈីលីសម្រាប់លីចូម) ហើយត្រូវបង្កើតតារាងពិន្ទុហានិភ័យនយោបាយដោយប្រើប្រាស់ចំណាត់ថ្នាក់ស្តង់ដារ៖ ស្ថិរភាពរបបនយោបាយ, ភាពអាចទស្សន៍ទាយបាននៃបទប្បញ្ញត្តិ, ប្រវត្តិនៃការរឹបអូសទ្រព្យសម្បត្តិ, ការប្រឈមនឹងទណ្ឌកម្ម, និងប្រតិទិនបោះឆ្នោត។ ក្រុមការងារបង្ហាញតារាងពិន្ទុរបស់ពួកគេ និងយុទ្ធសាស្ត្រកាត់បន្ថយហានិភ័យដែលបានណែនាំទៅកាន់ថ្នាក់រៀន ដូចជាការរាយការណ៍សង្ខេបជូនគណៈកម្មាធិការវិវិយោគសាជីវកម្ម។
- **ការពិសោធន៍ចរចាគោលនយោបាយពាណិជ្ជកម្ម (Trade Policy Negotiation Simulation)** — គូនីមួយៗតំណាងឱ្យប្រទេសពីរនៅក្នុងវិវាទ WTO សន្មតលើកម្មវិធីឧបត្ថម្ភធនថាមពលកកើតឡើងវិញ។ គូនីមួយៗទទួលបានព័ត៌មានសង្ខេបដែលមានផលប្រយោជន៍សេដ្ឋកិច្ច, កាតព្វកិច្ច WTO, និងដែនកំណត់នយោបាយរបស់ប្រទេសពួកគេ។ ពួកគេត្រូវតែចរចាស្វែងរកដំណោះស្រាយដែលភាគីទាំងពីរអាចទទួលយកបាន, កត់ត្រាវាជាសេចក្តីថ្លែងការណ៍ព្រមព្រៀងជាផ្លូវការ, ហើយបន្ទាប់មកបង្ហាញវាទៅកាន់ថ្នាក់រៀនសម្រាប់ការបោះឆ្នោតថាតើវាស្របតាមច្បាប់ WTO ដែរឬទេ។

### ករណីសិក្សាដែលត្រូវបានប្រើប្រាស់ (Case Studies Used)
- **សង្គ្រាមពាណិជ្ជកម្មសហរដ្ឋអាមេរិក-ចិន (២០១៨–បច្ចុប្បន្ន)** — និស្សិតវិភាគលើការកើនឡើងនៃពន្ធគយរវាងសហរដ្ឋអាមេរិក និងចិនតាមរយៈកែវភ្នែកនៃទ្រឹស្តីតថនិយម (រដ្ឋនីមួយៗស្វែងរកផលប្រយោជន៍ជាតិ) និងទ្រឹស្តីសេរីនិយម (ដែនកំណត់ស្ថាប័នពី WTO) ដោយពិនិត្យមើលពីរបៀបដែលសង្គ្រាមពាណិជ្ជកម្មរៀបចំរចនាសម្ព័ន្ធខ្សែសង្វាក់ផ្គត់ផ្គង់សកលឡើងវិញនៅក្នុងវិស័យអេឡិចត្រូនិក និងផ្ទាំងសូឡា និងអ្វីដែលករណីនេះលាតត្រដាងអំពីដែនកំណត់នៃការអនុវត្តច្បាប់របស់ WTO។
- **យន្តការកែសម្រួលព្រំដែនកាបូនរបស់សហភាពអឺរ៉ុប (The EU Carbon Border Adjustment Mechanism)** — និស្សិតពិនិត្យមើល EU CBAM ជាករណីជាក់ស្តែងដែលគោលនយោបាយពាណិជ្ជកម្ម, គោលនយោបាយអាកាសធាតុ, និងច្បាប់របស់ WTO ប៉ះទង្គិចគ្នា — ដោយវិភាគលើអាគុយម៉ង់ផ្លូវច្បាប់សម្រាប់ និងប្រឆាំងនឹងភាពស៊ីគ្នារបស់ CBAM ទៅនឹងច្បាប់ WTO, ផលប៉ះពាល់ការបែងចែកធនធានលើប្រទេសកំពុងអភិវឌ្ឍន៍ដែលនាំចេញ, និងអ្វីដែលឧបករណ៍នេះលាតត្រដាងអំពីរបៀបដែលស្ថាប័ននយោបាយរៀបចំសេដ្ឋកិច្ចនៃការផ្លាស់ប្តូរនិរន្តរភាព។
- **ហានិភ័យរដ្ឋប្រហារ និងការវិនិយោគលើវិស័យធ្វើរ៉ែនៅអាហ្វ្រិកខាងលិច** — និស្សិតពិនិត្យមើលពីរបៀបដែលរដ្ឋប្រហារនៅអាហ្វ្រិកខាងលិចមួយចំនួន (ម៉ាលី ២០២១, ហ្គីណេ ២០២១, បួគីណាហ្វាសូ ២០២២, នីហ្សេ ២០២៣) ប៉ះពាល់ដល់ប្រតិបត្តិការធ្វើរ៉ែមាស និងបុកស៊ីត (Bauxite) ដែលជាកម្មសិទ្ធិរបស់ក្រុមហ៊ុនពហុជាតិ ដោយវិភាគពីរបៀបដែលហានិភ័យនយោបាយត្រូវបានកំណត់ថ្លៃមុន និងក្រោយព្រឹត្តិការណ៍នីមួយៗ និងថាតើបរាជ័យនៃការត្រួតពិនិត្យហានិភ័យណាខ្លះដែលបន្សល់ទុកឱ្យក្រុមហ៊ុនងាយរងគ្រោះ។

### ការពិសោធន៍ និងការដើរតួសម្តែង (Simulations & Role-Play)
- **លំហាត់រៀបចំផែនការសេណារីយ៉ូនយោបាយ (Political Scenario Planning Exercise)** — និស្សិតម្នាក់ៗត្រូវបានចាត់តាំងជាអ្នកវិភាគយុទ្ធសាស្ត្រសម្រាប់ក្រុមហ៊ុនពហុជាតិសន្មតដែលមានប្រតិបត្តិការនៅក្នុងប្រទេសចំនួនប្រាំ។ ពួកគេទទួលបានព័ត៌មានសង្ខេបអំពីការបោះឆ្នោតនាពេលខាងមុខនៅក្នុងប្រទេសមួយ, ការសម្រេចចិត្តលើទណ្ឌកម្មដែលកំពុងរង់ចាំនៅក្នុងប្រទេសមួយទៀត, និងការកែសម្រួលបទប្បញ្ញត្តិនៅក្នុងប្រទេសទីបី។ និស្សិតត្រូវតែសរសេរអនុស្សរណៈរៀបចំផែនការសេណារីយ៉ូមួយទំព័រដែលកំណត់អត្តសញ្ញាណហានិភ័យនយោបាយដែលមានឥទ្ធិពលបំផុតពីរបចំពោះប្រតិបត្តិការរបស់ក្រុមហ៊ុនក្នុងរយៈពេល ១៨ ខែខាងមុខ និងណែនាំយុទ្ធសាស្ត្រឆ្លើយតបសម្រាប់ហានិភ័យនីមួយៗ។
- **ការជជែកដេញដោលរវាងស្ថាបនានិយម ធៀបនឹង តថនិយម (Constructivism vs. Realism Debate)** — ថ្នាក់រៀនត្រូវបានបែងចែកជាភាគីតថនិយម និងស្ថាបនានិយម ហើយត្រូវបានចាត់តាំងឱ្យពន្យល់ពីព្រឹត្តិការណ៍ប្រវត្តិសាស្ត្រដូចគ្នា (ឧទាហរណ៍ ជោគជ័យយ៉ាងឆាប់រហ័សនៃពិធីសារម៉ុងរ៉េអាល់) ពីទស្សនៈទ្រឹស្តីនីមួយៗ។ ភាគីនីមួយៗត្រូវតែធ្វើការបង្ហាញឱ្យបានល្អបំផុត ហើយបន្ទាប់មកថ្នាក់រៀនបូកសរុបឡើងវិញថាតើទ្រឹស្តីណាដែលពន្យល់ពីភស្តុតាងបានច្រើនជាង និងកន្លែងណាដែលក្របខ័ណ្ឌការងារនីមួយៗដួលរលំ — បណ្តុះទម្លាប់នៃការប្រើប្រាស់ទ្រឹស្តីជាកែវភ្នែកវិភាគ ជាជាងការអះអាងអំពីការពិត។

### វិធីសាស្ត្រវាយតម្លៃ (Assessment Methods)
- **កំណត់ហេតុវិភាគព្រឹត្តិការណ៍បច្ចុប្បន្ន (Current Events Analysis Journal)** — ការសរសេរកំណត់ហេតុចំនួនប្រាំកម្រិត ២៥០ ពាក្យ (២០% នៃពិន្ទុ) ដាក់ជូនរៀងរាល់ពីរសប្តាហ៍ម្តង ដោយភ្ជាប់ព្រឹត្តិការណ៍ភូមិសាស្ត្រនយោបាយ ឬពាណិជ្ជកម្មបច្ចុប្បន្នទៅនឹងក្របខ័ណ្ឌវគ្គសិក្សាជាក់លាក់; ដាក់ពិន្ទុលើភាពត្រឹមត្រូវនៃការអនុវត្តក្របខ័ណ្ឌការងារ និងគុណភាពនៃការតភ្ជាប់វិភាគ ជាជាងលើទស្សនៈនយោបាយ។
- **របាយការណ៍ហានិភ័យនយោបាយកម្រិតប្រទេស (Political Risk Country Report)** — គម្រោងតារាងពិន្ទុហានិភ័យភូមិសាស្ត្រនយោបាយជាក្រុម (២៥% នៃពិន្ទុ) វាយតម្លៃលើភាពទូលំទូលាយនៃចំណាត់ថ្នាក់ហានិភ័យ, គុណភាពនៃភស្តុតាងដែលបានប្រើប្រាស់, និងលក្ខណៈសមហេតុផលនៃយុទ្ធសាស្ត្រកាត់បន្ថយហានិភ័យដែលបានណែនាំ។
- **ការប្រឡងពាក់កណ្តាលឆមាស (Midterm Exam)** — ការប្រឡងរយៈពេល ៩០ នាទី (២៥% នៃពិន្ទុ) គ្របដណ្តប់លើទម្រង់នៃរដ្ឋាភិបាល, ទ្រឹស្តីទំនាក់ទំនងអន្តរជាតិ, និងអង្គការអន្តរជាតិ រួមមានសំណួរចម្លើយខ្លីៗ, ការវិភាគរដ្ឋាភិបាលប្រៀបធៀបមួយ, និងតែងសេចក្តីមួយដែលអនុវត្តទ្រឹស្តីទំនាក់ទំនងអន្តរជាតិពីរទៅលើករណីដែលបានផ្តល់ឱ្យ។
- **សារណាស្រាវជ្រាវចុងក្រោយ (Final Research Paper)** — ក្រដាសស្រាវជ្រាវសរសេរផ្ទាល់ខ្លួនកម្រិត ១.៥០០ ពាក្យ (៣០% នៃពិន្ទុ) វិភាគលើហានិភ័យនយោបាយជាក់ស្តែងមួយដែលប្រឈមនឹងវិស័យពហុជាតិជាក់លាក់ (ថាមពល, ធ្វើរ៉ែ, កសិកម្ម, ឬកម្មន្តសាលបច្ចេកវិទ្យា) ដោយផ្អែកលើទ្រឹស្តីវិទ្យាសាស្ត្រនយោបាយ, ភស្តុតាងករណីសិក្សាកម្រិតប្រទេសជាក់លាក់, និងការណែនាំជាក់ស្តែងសម្រាប់របៀបដែលក្រុមហ៊ុនគួរគ្រប់គ្រងហានិភ័យនោះ។

### របៀបសិក្សាដោយខ្លួនឯងជាមួយ AI (How to Self-Learn This with AI)
១. **អនុវត្តទ្រឹស្តីទំនាក់ទំនងអន្តរជាតិទៅលើព្រឹត្តិការណ៍បច្ចុប្បន្ន** — សួរ AI៖ "សូមពិពណ៌នាអំពីជម្លោះភូមិសាស្ត្រនយោបាយ ឬវិវាទពាណិជ្ជកម្មបច្ចុប្បន្នមួយមកកាន់ខ្ញុំ។ ខ្ញុំនឹងពន្យល់វាដោយប្រើប្រាស់ទ្រឹស្តីតថនិយម (Realism)។ បន្ទាប់មក សួរខ្ញុំឱ្យពន្យល់ពីព្រឹត្តិការណ៍ដដែលនោះពីទស្សនៈសេរីនិយម (Liberalism) និងចុងក្រោយពីទស្សនៈស្ថាបនានិយម (Constructivism)។ បន្ទាប់ពីការពន្យល់នីមួយៗ សូមចង្អុលបង្ហាញកន្លែងដែលការអនុវត្តទ្រឹស្តីរបស់ខ្ញុំមានភាពខ្លាំង និងកន្លែងដែលវាដួលរលំ។"

៣. **ពិសោធន៍ការរាយការណ៍សង្ខេបអនុលោមភាពទណ្ឌកម្ម** — សួរ AI៖ "ខ្ញុំជាមន្ត្រីអនុលោមភាពថ្មីម្នាក់នៅក្រុមហ៊ុនជួញដូរមួយ។ ណែនាំខ្ញុំអំពីសេណារីយ៉ូមួយដែលអ្នកផ្គត់ផ្គង់ដែលបានស្នើឡើងរបស់យើងនៅក្នុងប្រទេសទីបីអាចមានតំណភ្ជាប់ដោយប្រយោលទៅកាន់អង្គភាពដែលរងទណ្ឌកម្ម។ សួរខ្ញុំនូវសំណួរដែលមន្ត្រីអនុលោមភាពគួរចោទសួរ, ពន្យល់ពីការប្រឈមផ្លូវច្បាប់នៅជំហាននីមួយៗ, និងជួយខ្ញុំសម្រេចចិត្តថាតើត្រូវបន្តដំណើរការដែរឬទេ។"
